from collections import Counter
x = covid.age_bracket

cou = Counter(x)
d=dict(cou)

key=list(d.keys())
value=list(d.values())

for i in key:
    if(str(i) == 'nan'):
        n = key.index(i)
        del key[n] # removal of empty cells
        del value[n] # removal of empty cells
    if(i == '28-35'):
        n = key.index(i)
        key.pop(n)
        value.pop(n)

for i in range(len(key)):
    key[i] = int(key[i])

for i in range (len(key)): # sorting in ascending order
    for j in range(i + 1, len(key)):        
        if(key[i] > key[j]):
            temp = key[i]
            key[i] = key[j]
            key[j] = temp
            temp1 = value[i]
            value[i] = value[j]
            value[j] = temp1

print(d)
